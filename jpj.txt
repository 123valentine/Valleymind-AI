function resolveApiBaseUrl() {
    if (window.VALLEYMIND_API_BASE_URL) return window.VALLEYMIND_API_BASE_URL.replace(/\/$/, "");
    var localHosts = new Set(["localhost", "127.0.0.1", "0.0.0.0"]);
    var isLocal = localHosts.has(window.location.hostname);
    var isBackendPort = window.location.port === "8000";
    if (window.location.protocol === "file:" || (isLocal && !isBackendPort)) return "http://127.0.0.1:8000";
    return window.location.origin;
}

var API_BASE_URL = resolveApiBaseUrl();
var AUTH_TOKEN_KEY = "valleymind_session_token";

function apiUrl(path) { return API_BASE_URL + path; }

function networkErrorMessage(err, url) {
    var reason = err && err.message ? err.message : String(err || "unknown error");
    if ((err && err.name === "AbortError") || /aborted/i.test(reason))
        return "That answer is taking longer than usual. Please try a shorter version, or send it again and I'll continue from there.";
    return "Network error reaching backend at " + url + ": " + reason;
}

async function apiFetch(path, options) {
    options = options || {};
    var url = apiUrl(path);
    var controller = new AbortController();
    var timeout = window.setTimeout(function () { controller.abort(); }, options.timeoutMs || 30000);
    try {
        return await fetch(url, Object.assign({}, options, { signal: controller.signal }));
    } catch (err) {
        console.error("[PhoneStudio network]", { url: url, apiBaseUrl: API_BASE_URL, pageOrigin: window.location.origin, message: err && err.message ? err.message : String(err) });
        throw new Error(networkErrorMessage(err, url));
    } finally {
        window.clearTimeout(timeout);
    }
}

function getAuthToken() { return sessionStorage.getItem(AUTH_TOKEN_KEY) || ""; }
function setAuthToken(token) { if (token) sessionStorage.setItem(AUTH_TOKEN_KEY, token); }
function clearAuthToken() { sessionStorage.removeItem(AUTH_TOKEN_KEY); }

function authHeaders(extra) {
    extra = extra || {};
    var headers = Object.assign({}, extra);
    var token = getAuthToken();
    if (token) { headers["X-Session-Token"] = token; headers["Authorization"] = "Bearer " + token; }
    return headers;
}

async function phoneLogin(event) {
    if (event) event.preventDefault();
    var errorEl = document.getElementById("phoneLoginError");
    if (errorEl) errorEl.textContent = "";
    var email = document.getElementById("phoneLoginEmail").value.trim();
    var password = document.getElementById("phoneLoginPassword").value;
    if (!email) { if (errorEl) errorEl.textContent = "Enter your email."; return; }
    if (!password) { if (errorEl) errorEl.textContent = "Enter your password."; return; }

    setConnStatus("connecting");

    try {
        var res = await apiFetch("/auth/login", {
            method: "POST",
            credentials: "include",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ email: email, password: password })
        });
        var data = await res.json();
        if (!res.ok || data.status !== "success") {
            throw new Error(data.message || "Login failed");
        }
        setAuthToken(data.session_token || "");
        var ok = await phoneCheckSession();
        if (ok) {
            localStorage.setItem('isLoggedIn', 'true');
            hidePhoneLogin();
            showPhoneToast("Connected. Memory online.", false);
            if (errorEl) errorEl.textContent = "";
            document.getElementById("phoneLoginEmail").value = "";
            document.getElementById("phoneLoginPassword").value = "";
        } else {
            throw new Error("Session validation failed");
        }
    } catch (err) {
        setConnStatus("offline");
        if (errorEl) errorEl.textContent = err.message || "Login failed";
        showPhoneToast(err.message || "Login failed");
    }
}

    currentChatId = chatId;
    await phoneLoadChatHistory(chatId);
    phoneRenderSessionList();
    closePhoneSidebar();
    // Update header title
    var titleEl = document.getElementById("phoneChatTitle");
    if (titleEl) {
        var found = sessionList.find(function (s) { return s.chat_id === chatId; });
        titleEl.textContent = (found && found.title) || "New Thread";
    }
    phoneFocusInput();
}

async function phoneLoadChatHistory(chatId) {
    try {
        var params = chatId ? "?chat_id=" + encodeURIComponent(chatId) : "";
        var res = await apiFetch("/chat/history" + params, { credentials: "include", headers: authHeaders() });
        if (!res.ok) return false;
        var data = await res.json();
        if (data.status === "success") { phoneRenderMessages(data.messages || []); return true; }
    } catch (err) { console.warn("Chat history unavailable.", err); }
    return false;
}

async function phoneLoadSessions() {
    try {
        var res = await apiFetch("/chat/sessions", { credentials: "include", headers: authHeaders() });
        if (!res.ok) return;
        var data = await res.json();
        if (data.status === "success") { sessionList = data.sessions || []; phoneRenderSessionList(); }
    } catch (err) { console.warn("Failed to load sessions.", err); }
}

function phoneRenderSessionList() {
    var list = document.getElementById("phoneHistoryList");
    if (!list) return;
    list.innerHTML = "";
    var sessions = Array.isArray(sessionList) ? sessionList : [];
    var visible = sessions.filter(function (s) {
        var title = (s.title || "").trim();
        var isGeneric = title === "" || title === "New Chat" || title === "Untitled Thread";
        var isEmpty = !s.message_count || s.message_count === 0;
        if (isGeneric && isEmpty && s.chat_id !== currentChatId) return false;
        return true;
    });
    var countEl = document.getElementById("phoneSessionCount");
    if (countEl) countEl.textContent = visible.length + " session" + (visible.length !== 1 ? "s" : "");
    if (visible.length === 0) {
        list.innerHTML = '<div style="padding:10px 6px;color:#475569;font-size:10px;text-align:center;">No previous sessions</div>';
        return;
    }
    var pinnedIds = phoneGetPinnedIds();
    var sorted = visible.slice().sort(function (a, b) {
        var aP = pinnedIds.indexOf(a.chat_id) !== -1 ? 1 : 0;
        var bP = pinnedIds.indexOf(b.chat_id) !== -1 ? 1 : 0;
        return bP - aP;
    });
    sorted.forEach(function (session) {
        var isPinned = pinnedIds.indexOf(session.chat_id) !== -1;
        var isActive = session.chat_id === currentChatId;
        var item = document.createElement("div");
        item.style.cssText = "display:flex;justify-content:space-between;align-items:center;width:100%;position:relative;padding:8px 10px;border-radius:8px;cursor:pointer;transition:all 0.15s;background:" + (isActive ? "rgba(0,212,255,0.08)" : "transparent") + ";border-left:" + (isActive ? "3px solid #00d4ff" : "3px solid transparent") + ";";
        item.onclick = function () { phoneSwitchSession(session.chat_id); };
        var leftSpan = document.createElement("span");
        leftSpan.style.cssText = "display:flex;align-items:center;gap:6px;flex:1;overflow:hidden;";
        if (isPinned) {
            var pinBadge = document.createElement("span");
            pinBadge.textContent = "📌";
            pinBadge.style.cssText = "font-size:11px;flex-shrink:0;";
            leftSpan.appendChild(pinBadge);
        }
        var text = document.createElement("span");
        text.style.cssText = "font-size:12px;color:" + (isActive ? "#00d4ff" : "#cbd5e1") + ";font-weight:500;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;";
        text.textContent = session.title || "Untitled Thread";
        leftSpan.appendChild(text);
        var dotsBtn = document.createElement("button");
        dotsBtn.innerHTML = '<i data-lucide="more-vertical" style="width:14px;height:14px;"></i>';
        dotsBtn.title = "More options";
        dotsBtn.style.cssText = "background:none;border:none;cursor:pointer;padding:4px;border-radius:6px;color:#64748b;display:flex;align-items:center;touch-action:manipulation;flex-shrink:0;";
        dotsBtn.dataset.sessionId = session.chat_id;
        dotsBtn.onclick = function (e) {
            e.stopPropagation();
            phoneOpenSessionCtxMenu(e, session);
        };
        item.appendChild(leftSpan);
        item.appendChild(dotsBtn);
        list.appendChild(item);
    });
    if (typeof lucide !== "undefined" && lucide.createIcons) lucide.createIcons();
}

function phoneRenamePrompt(session) {
    if (!session || !session.chat_id) return;
    var curTitle = session.title || "";
    var newTitle = prompt("Rename session:", curTitle);
    if (newTitle && newTitle.trim() !== "" && newTitle.trim() !== curTitle) {
        apiFetch("/chat/session/rename", {
            method: "POST",
            headers: authHeaders({ "Content-Type": "application/json" }),
            body: JSON.stringify({ chat_id: session.chat_id, title: newTitle.trim() })
        }).then(function (res) {
            if (res.ok) { phoneLoadSessions(); return res.json(); }
        }).then(function () {
            if (session.chat_id === currentChatId) {
                var titleEl = document.getElementById("phoneChatTitle");
                if (titleEl) titleEl.textContent = newTitle.trim();
            }
            showPhoneToast("Session renamed.", false);
        }).catch(function () {
            showPhoneToast("Rename failed.");
        });
    }
}

function phoneShareThreadFromId(chatId) {
    if (!chatId) { showPhoneToast("No thread to share."); return; }
    var shareUrl = window.location.origin + "/share/" + chatId;
    navigator.clipboard.writeText(shareUrl).then(function () {
        showPhoneToast("Share link copied!", false);
    }).catch(function () {
        showPhoneToast("Could not copy link.");
    });
}

function phoneSessionCtxDo(action) {
    var session = _sessionCtxTarget;
    phoneCloseSessionCtxMenu();
    if (!session) return;
    if (action === "pin") { phoneTogglePinSession(session.chat_id); phoneRenderSessionList(); }
    else if (action === "rename") { phoneRenamePrompt(session); }
    else if (action === "share") { phoneShareThreadFromId(session.chat_id); }
    else if (action === "delete") { if (confirm("Delete this thread?")) phoneDeleteSession(session.chat_id); }
}

function phoneRenderMessages(messages) {
    phoneClearMessages();
    messageStore.length = 0;
    if (!Array.isArray(messages) || messages.length === 0) return;
    messages.forEach(function (message) {
        var role = message.role === "assistant" ? "ai" : "user";
        phoneAddMessage(message.content || "", role, message.image_data || null);
        if (message.reaction) { var idx = messageStore.length - 1; messageStore[idx].reaction = message.reaction; }
    });
    setTimeout(function () {
        document.querySelectorAll('#messages .msg-toolbar').forEach(function (toolbar) {
            var upBtn = toolbar.querySelector('[data-reaction="up"]');
            var downBtn = toolbar.querySelector('[data-reaction="down"]');
            if (!upBtn || !downBtn) return;
            var idx = parseInt(upBtn.dataset.index, 10);
            var msg = messageStore[idx];
            if (!msg || !msg.reaction) return;
            if (msg.reaction === "up") upBtn.classList.add("up-active");
            else if (msg.reaction === "down") downBtn.classList.add("down-active");
        });
        phoneRestorePinBadges();
    }, 50);
}

function phoneRestorePinBadges() {
    var pinnedIds = phoneGetPinnedMsgIds();
    if (pinnedIds.length === 0) return;
    var msgs = document.querySelectorAll('#messages .assistant-message');
    msgs.forEach(function (el, idx) {
        var toolbarBtns = el.querySelectorAll('button');
        var msgIdx = -1;
        for (var j = 0; j < toolbarBtns.length; j++) {
            var di = toolbarBtns[j].dataset.index;
            if (di !== undefined) { msgIdx = parseInt(di, 10); break; }
        }
        if (msgIdx !== -1 && pinnedIds.indexOf(String(msgIdx)) !== -1) {
            var bub = el.querySelector('.bubble');
            if (bub && !bub.querySelector('.msg-pin-badge')) {
                var badge = document.createElement("span");
                badge.className = "msg-pin-badge";
                badge.textContent = " 📌";
                badge.style.cssText = "font-size:11px;margin-left:6px;";
                bub.appendChild(badge);
            }
        }
    });
}

function phoneToggleReaction(btn, msgIndex, reaction) {
    var toolbar = btn.parentElement;
    var upBtn = toolbar.querySelector('[data-reaction="up"]');
    var downBtn = toolbar.querySelector('[data-reaction="down"]');
    var isActive = btn.classList.contains("up-active") || btn.classList.contains("down-active");
    var newReaction = isActive ? "" : reaction;
    upBtn.classList.remove("up-active", "down-active");
    downBtn.classList.remove("up-active", "down-active");
    if (newReaction === "up") upBtn.classList.add("up-active");
    else if (newReaction === "down") downBtn.classList.add("down-active");
    messageStore[msgIndex].reaction = newReaction || null;
    if (currentChatId) {
        apiFetch("/chat/sessions/" + encodeURIComponent(currentChatId) + "/reaction", { method: "POST", credentials: "include", headers: authHeaders({ "Content-Type": "application/json" }), body: JSON.stringify({ message_index: msgIndex, reaction: newReaction }) }).catch(function (err) { console.warn("Failed to save reaction:", err); });
    }
}

function phoneCopyMessage(btn) {
    var wrapper = btn.closest('.assistant-message');
    var bubble = wrapper ? wrapper.querySelector('.bubble') : null;
    var text = bubble ? bubble.textContent : '';
    if (!text) { showPhoneToast("Nothing to copy."); return; }
    function doCopy(cb) {
        navigator.clipboard.writeText(text).then(cb).catch(function () {
            var ta = document.createElement("textarea"); ta.value = text;
            document.body.appendChild(ta); ta.select(); document.execCommand("copy"); document.body.removeChild(ta); cb();
        });
    }
    doCopy(function () {
        var icon = btn.querySelector('i');
        if (icon) { icon.setAttribute('data-lucide', 'check'); icon.style.color = '#00ff88'; }
        btn.style.color = '#00ff88';
        showPhoneToast("Copied to clipboard.", false);
        setTimeout(function () {
            if (icon) { icon.setAttribute('data-lucide', 'copy'); icon.style.color = ''; }
            btn.style.color = '';
            if (typeof lucide !== 'undefined' && lucide.createIcons) lucide.createIcons();
        }, 1500);
        if (typeof lucide !== 'undefined' && lucide.createIcons) lucide.createIcons();
    });
}

function phoneToggleVoicePlayback(btn) {
    var icon = btn.querySelector('i');
    var isPlaying = btn.dataset.playing === 'true';
    if (isPlaying && window.speechSynthesis.speaking) {
        window.speechSynthesis.pause(); btn.dataset.playing = 'false';
        if (icon) icon.setAttribute('data-lucide', 'play');
        if (typeof lucide !== 'undefined' && lucide.createIcons) lucide.createIcons();
        return;
    }
    if (window.speechSynthesis.paused) {
        window.speechSynthesis.resume(); btn.dataset.playing = 'true';
        if (icon) icon.setAttribute('data-lucide', 'pause');
        if (typeof lucide !== 'undefined' && lucide.createIcons) lucide.createIcons();
        return;
    }
    var wrapper = btn.closest('.assistant-message');
    var bubble = wrapper ? wrapper.querySelector('.bubble') : null;
    var text = bubble ? bubble.textContent : '';
    if (!text) return;
    window.speechSynthesis.cancel();
    var utterance = new SpeechSynthesisUtterance(text);
    utterance.rate = 1.0; utterance.pitch = 0.85;
    utterance.onend = function () {
        btn.dataset.playing = 'false';
        if (icon) icon.setAttribute('data-lucide', 'play');
        if (typeof lucide !== 'undefined' && lucide.createIcons) lucide.createIcons();
    };
    window.speechSynthesis.speak(utterance);
    btn.dataset.playing = 'true';
    if (icon) icon.setAttribute('data-lucide', 'pause');
    if (typeof lucide !== 'undefined' && lucide.createIcons) lucide.createIcons();
}

function browserSpeak(text) {
    if (!("speechSynthesis" in window)) return;
    window.speechSynthesis.cancel();
    var utterance = new SpeechSynthesisUtterance(text);
    var voices = window.speechSynthesis.getVoices();
    var selectedVoice = voices.find(function (v) { return v.name.includes("Microsoft David"); });
    if (!selectedVoice) selectedVoice = voices.find(function (v) { return v.name.toLowerCase().includes("male") && v.lang.startsWith("en"); });
    if (!selectedVoice) selectedVoice = voices.find(function (v) { return v.name.includes("Google US English"); });
    if (!selectedVoice) selectedVoice = voices.find(function (v) { return v.lang.startsWith("en-"); });
    if (selectedVoice) utterance.voice = selectedVoice;
    utterance.rate = 1.0; utterance.pitch = 0.85;
    window.speechSynthesis.speak(utterance);
}

function phoneGetMessageText(msgIndex) {
    if (msgIndex < 0 || msgIndex >= messageStore.length) return "";
    var msg = messageStore[msgIndex];
    return msg ? msg.content : "";
}

function phoneCtxDo(action) {
    var msgIndex = _phoneCtxMsgIndex;
    phoneCloseCtxMenu();

    if (action === "draft") {
        var txt = phoneGetMessageText(msgIndex);
        if (!txt) { showPhoneToast("No message content to draft."); return; }
        if ("speechSynthesis" in window) {
            window.speechSynthesis.cancel();
            var utterance = new SpeechSynthesisUtterance(txt);
            utterance.rate = 0.9; utterance.pitch = 0.85;
            var voices = window.speechSynthesis.getVoices();
            var voice = voices.find(function (v) { return v.lang.startsWith("en-"); });
            if (voice) utterance.voice = voice;
            window.speechSynthesis.speak(utterance);
            showPhoneToast("Response Draft — voice synthesis active.", false);
        } else {
            showPhoneToast("Response Draft — layout compiled (" + txt.length + " chars).", false);
        }
        return;
    }

    if (action === "email") {
        var txt = phoneGetMessageText(msgIndex);
        if (!txt) { showPhoneToast("No message content to email."); return; }
        var subject = encodeURIComponent("Valley Mind-AI Response Draft");
        var body = encodeURIComponent(txt);
        window.open("mailto:?subject=" + subject + "&body=" + body, "_blank");
        showPhoneToast("Email draft opened in default client.", false);
        return;
    }

    if (action === "detail") {
        var msg = messageStore[msgIndex];
        if (!msg) { showPhoneToast("Message not found."); return; }
        var detail = "Message #" + msgIndex + "\nRole: " + msg.role + "\nLength: " + (msg.content || "").length + " chars\nPinned: " + (phoneIsMsgPinned(msgIndex) ? "Yes" : "No") + "\n---\n" + (msg.content || "").slice(0, 200);
        showPhoneToast("Detail logged to console.");
        console.log("[Response Details]", { index: msgIndex, role: msg.role, length: msg.content.length, content: msg.content.slice(0, 500) });
        return;
    }

    if (action === "report") {
        var msg = messageStore[msgIndex];
        if (!msg) { showPhoneToast("Message not found."); return; }
        var reportData = { index: msgIndex, role: msg.role, content: (msg.content || "").slice(0, 1000), chat_id: currentChatId, user_agent: navigator.userAgent };
        showPhoneToast("Report Issue — logging debug data.");
        console.log("[Report Issue]", reportData);
        apiFetch("/chat/report", {
            method: "POST",
            headers: authHeaders({ "Content-Type": "application/json" }),
            body: JSON.stringify(reportData)
        }).then(function (r) {
            if (r.ok) showPhoneToast("Issue reported to team.", false);
        }).catch(function () {});
        return;
    }

    if (action === "pin") {
        if (msgIndex < 0 || msgIndex >= messageStore.length) { showPhoneToast("No message selected."); return; }
        phoneToggleMsgPin(msgIndex);
        return;
    }

    if (action === "rename") {
        if (!currentChatId) { showPhoneToast("No active thread."); return; }
        var curTitle = "";
        var found = sessionList.find(function (s) { return s.chat_id === currentChatId; });
        if (found) curTitle = found.title || "";
        var newTitle = prompt("Modify session title:", curTitle);
        if (newTitle && newTitle.trim() !== "") {
            apiFetch("/chat/session/rename", {
                method: "POST",
                headers: authHeaders({ "Content-Type": "application/json" }),
                body: JSON.stringify({ chat_id: currentChatId, title: newTitle.trim() })
            }).then(function (res) {
                if (res.ok) { phoneLoadSessions(); return res.json(); }
            }).then(function () {
                var titleEl = document.getElementById("phoneChatTitle");
                if (titleEl) titleEl.textContent = newTitle.trim();
                showPhoneToast("Thread renamed.", false);
            }).catch(function () {
                showPhoneToast("Rename failed.");
            });
        }
        return;
    }

    if (action === "delete") {
        if (!currentChatId) { showPhoneToast("No active thread."); return; }
        if (!confirm("Delete this entire thread?")) return;
        phoneDeleteSession(currentChatId);
        return;
    }
}

async function phoneRegenerateMessage(msgIndex) {
    if (!currentChatId) { showPhoneToast("No active session."); return; }
    if (msgIndex < 0 || msgIndex >= messageStore.length) { showPhoneToast("Invalid message."); return; }
    var msg = messageStore[msgIndex];
    if (!msg || msg.role !== "assistant") { showPhoneToast("Can only regenerate AI responses."); return; }
    var prevMsg = msgIndex > 0 ? messageStore[msgIndex - 1] : null;
    if (!prevMsg || prevMsg.role !== "user") { showPhoneToast("No preceding user message to re-prompt."); return; }
    var promptText = prevMsg.content || "";
    if (!promptText) { showPhoneToast("Preceding message is empty."); return; }
    showPhoneToast("Regenerating...", false);

    var container = document.getElementById("messages");
    if (!container) return;
    var allMsgs = container.querySelectorAll(".assistant-message");
    var targetEl = null;
    for (var mi = 0; mi < allMsgs.length; mi++) {
        var idx = parseInt(allMsgs[mi].querySelector('[data-msgindex]') ? allMsgs[mi].querySelector('[data-msgindex]').dataset.msgindex : allMsgs[mi].dataset.msgindex, 10);
        if (isNaN(idx)) {
            var toolbarBtns = allMsgs[mi].querySelectorAll("button");
            for (var bi = 0; bi < toolbarBtns.length; bi++) {
                var di = toolbarBtns[bi].dataset.index;
                if (di !== undefined) { idx = parseInt(di, 10); break; }
            }
        }
        if (idx === msgIndex) { targetEl = allMsgs[mi]; break; }
    }
    if (!targetEl) { showPhoneToast("Could not locate message bubble."); return; }
    var bubbleDiv = targetEl.querySelector(".bubble");
    if (bubbleDiv) { bubbleDiv.textContent = "Regenerating..."; msg.content = ""; }

    try {
        var body = { message: promptText, chat_id: currentChatId };
        var streamRes = await fetch(apiUrl("/chat/stream"), {
            method: "POST", credentials: "include",
            headers: authHeaders({ "Content-Type": "application/json" }),
            body: JSON.stringify(body)
        });
        if (!streamRes.ok) {
            if (streamRes.status === 401) throw new Error("Session expired.");
            throw new Error("Stream returned " + streamRes.status);
        }
        var reader = streamRes.body.getReader();
        var decoder = new TextDecoder("utf-8");
        var buffer = "";
        var fullReply = "";
        while (true) {
            var chunk = await reader.read();
            if (chunk.done) break;
            buffer += decoder.decode(chunk.value, { stream: true });
            var lines = buffer.split("\n");
            buffer = lines.pop() || "";
            for (var li = 0; li < lines.length; li++) {
                var line = lines[li].trim();
                if (!line || !line.startsWith("data: ")) continue;
                var dataStr = line.slice(6).trim();
                if (!dataStr || dataStr === "{}" || dataStr === "null") continue;
                try { var evt = JSON.parse(dataStr); } catch (e) { continue; }
                if (evt.error) { throw new Error(evt.error); }
                if (evt.token !== undefined && evt.token !== "" && evt.token !== " ") {
                    var tok = String(evt.token);
                    if (tok.startsWith("{") || tok.startsWith("[")) continue;
                    fullReply += tok;
                    if (bubbleDiv) bubbleDiv.textContent = fullReply;
                    container.scrollTop = container.scrollHeight;
                }
            }
        }
        if (fullReply) {
            msg.content = fullReply;
            showPhoneToast("Response regenerated.", false);
            if (!isVoiceMuted) browserSpeak(fullReply);
            phoneLoadSessions();
        } else {
            if (bubbleDiv) bubbleDiv.textContent = "[Empty response]";
            showPhoneToast("Regeneration returned empty.");
        }
    } catch (err) {
        if (bubbleDiv) bubbleDiv.textContent = msg.content || "[Regeneration failed]";
        showPhoneToast("Regeneration error: " + (err.message || ""));
    }
}

// ── MAIN SEND: live streaming from Flask backend ──
window.phoneSend = async function () {
    if (!authCheckComplete || isThinking) return;
    var input = document.getElementById("phoneChatInput");
    var text = input.value.trim();
    if (!text) return;

    phoneSetThinking(true);

    if (!currentChatId) {
        try {
            var createRes = await apiFetch("/chat/sessions", { method: "POST", credentials: "include", headers: authHeaders({ "Content-Type": "application/json" }), body: JSON.stringify({ title: "New Chat" }) });
            if (createRes.ok) { var createData = await createRes.json(); if (createData.status === "success") currentChatId = createData.session.chat_id; }
        } catch (err) { console.warn("Failed to auto-create session.", err); }
    }

    phoneAddMessage(text, "user", _pendingImageData || null);
    input.value = "";
    var thinking = phoneAddThinkingIndicator();
    var streamError = null;

    try {
        var body = { message: text };
        if (currentChatId) body.chat_id = currentChatId;
        if (_pendingImageData) { body.image = _pendingImageData; _pendingImageData = null; var pc = document.getElementById("phoneImagePreview"); if (pc) pc.classList.remove("show"); }

        var streamRes = await fetch(apiUrl("/chat/stream"), {
            method: "POST", credentials: "include",
            headers: authHeaders({ "Content-Type": "application/json" }),
            body: JSON.stringify(body)
        });

        if (!streamRes.ok) {
            if (streamRes.status === 401) throw new Error("Session expired. Please log in again.");
            throw new Error("Stream endpoint returned " + streamRes.status);
        }

        var reader = streamRes.body.getReader();
        var decoder = new TextDecoder("utf-8");
        var buffer = "";
        var fullReply = "";
        var bubbleEl = null;
        var streamMeta = {};
        var statusIndicator = null;

        while (true) {
            var chunk = await reader.read();
            if (chunk.done) break;

            buffer += decoder.decode(chunk.value, { stream: true });

            var stripped = "";
            var rawLines = buffer.split("\n");
            for (var li = 0; li < rawLines.length; li++) {
                var rl = rawLines[li];
                if (rl.startsWith("data: ")) stripped += rl + "\n";
                else if (rl.length > 0 && !rl.startsWith("{") && !rl.startsWith("data:")) stripped += rl + "\n";
            }
            buffer = stripped;

            var lines = buffer.split("\n");
            buffer = lines.pop() || "";

            for (var j = 0; j < lines.length; j++) {
                var line = lines[j].trim();
                if (!line || !line.startsWith("data: ")) continue;
                var dataStr = line.slice(6).trim();
                if (!dataStr || dataStr === "{}" || dataStr === "null") continue;
                try { var evt = JSON.parse(dataStr); } catch (e) { continue; }

                if (evt.error) { streamError = evt.error; break; }

                if (evt.intent) {
                    if (!statusIndicator) {
                        phoneRemoveThinkingIndicator(thinking);
                        var statusDiv = document.createElement("div");
                        statusDiv.className = "assistant-message";
                        statusDiv.dataset.statusEl = "true";
                        var statusBubble = document.createElement("div");
                        statusBubble.className = "bubble";
                        statusBubble.style.cssText = "color:#94a3b8;font-style:italic;font-size:12px;";
                        statusBubble.textContent = "Searching the web...";
                        statusDiv.appendChild(statusBubble);
                        document.getElementById("messages").appendChild(statusDiv);
                        statusIndicator = statusDiv;
                        document.getElementById("messages").scrollTop = document.getElementById("messages").scrollHeight;
                    }
                    continue;
                }

                if (evt.token !== undefined) {
                    if (evt.token === "" || evt.token === " ") continue;
                    var tok = String(evt.token);
                    if (tok.startsWith("{") || tok.startsWith("[") || tok.indexOf('{"intent":') !== -1 || tok.indexOf('{"token":') !== -1 || tok.indexOf('{"done":') !== -1) continue;

                    if (statusIndicator) { statusIndicator.remove(); statusIndicator = null; }
                    if (!bubbleEl) {
                        phoneRemoveThinkingIndicator(thinking);
                        phoneAddMessage("", "ai");
                        var msgs = document.getElementById("messages");
                        bubbleEl = msgs ? msgs.lastElementChild : null;
                        if (bubbleEl) { var bub = bubbleEl.querySelector(".bubble"); if (bub) bub.textContent = ""; }
                    }
                    fullReply += tok;
                    if (bubbleEl) {
                        var bub = bubbleEl.querySelector(".bubble");
                        if (bub) {
                            bub.textContent = fullReply;
                            var cursor = document.createElement("span");
                            cursor.className = "stream-cursor";
                            cursor.style.cssText = "display:inline-block;width:2px;height:14px;background:#00d4ff;animation:blink 1s step-end infinite;vertical-align:text-bottom;margin-left:1px;";
                            bub.appendChild(cursor);
                            var container = document.getElementById("messages");
                            if (container) container.scrollTop = container.scrollHeight;
                        }
                    }
                }
                if (evt.done) {
                    if (statusIndicator) { statusIndicator.remove(); statusIndicator = null; }
                    streamMeta = evt;
                }
            }
            if (streamError) break;
        }

        if (!bubbleEl && !streamError) { phoneRemoveThinkingIndicator(thinking); if (fullReply) phoneAddMessage(fullReply, "ai"); }

        if (streamMeta.chat_id && !currentChatId) currentChatId = streamMeta.chat_id;

        if (streamMeta.updated_title && currentChatId) {
            var found = false;
            for (var i = 0; i < sessionList.length; i++) { if (sessionList[i].chat_id === currentChatId) { sessionList[i].title = streamMeta.updated_title; found = true; break; } }
            if (!found) sessionList.push({ chat_id: currentChatId, title: streamMeta.updated_title });
            phoneRenderSessionList();
        }
        phoneLoadSessions();
        if (!isVoiceMuted && fullReply) browserSpeak(fullReply);
    } catch (err) {
        if (!streamError) streamError = err.message || "Connection error";
    }

    // Fallback to non-streaming /chat
    if (streamError) {
        try {
            phoneRemoveThinkingIndicator(thinking);
            var body2 = { message: text };
            if (currentChatId) body2.chat_id = currentChatId;
            if (_pendingImageData) body2.image = _pendingImageData;
            var fallbackRes = await apiFetch("/chat", { method: "POST", credentials: "include", headers: authHeaders({ "Content-Type": "application/json" }), body: JSON.stringify(body2), timeoutMs: 120000 });
            var fallbackData = await fallbackRes.json();
            phoneRemoveThinkingIndicator(thinking);
            if (fallbackRes.status === 401) {
                clearAuthToken(); showPhoneLogin(); showPhoneToast("Session expired. Please log in again.");
                phoneSetThinking(false); return;
            }
            var reply = fallbackData.reply || fallbackData.message || "No response";
            if (fallbackData.chat_id && !currentChatId) currentChatId = fallbackData.chat_id;
            phoneAddMessage(reply, "ai");
            phoneLoadSessions();
        } catch (fallbackErr) {
            phoneRemoveThinkingIndicator(thinking);
            if (streamError.indexOf("401") !== -1 || streamError.indexOf("Session expired") !== -1) {
                clearAuthToken(); showPhoneLogin();
            } else {
                phoneAddMessage(streamError || fallbackErr.message || "Network error. Server may be starting up — please retry in a moment.", "ai");
                showPhoneToast(streamError || "Connection error — retrying...");
            }
        }
    }

    phoneSetThinking(false);
    phoneFocusInput();
    clearImageUpload();
};

// ── Auth session check ──
async function phoneCheckSession() {
    var wasComplete = authCheckComplete;
    authCheckComplete = false;

    if (!getAuthToken()) {
        if (localStorage.getItem('isLoggedIn') === 'true') {
            authCheckComplete = true;
            setConnStatus("online");
            hidePhoneLogin();
            phoneClearMessages();
            phoneLoadSessions();
            var stEl = document.getElementById("phoneStatusText");
            if (stEl) stEl.textContent = "Memory Online";
            return true;
        }
        authCheckComplete = true;
        showPhoneLogin();
        var stEl = document.getElementById("phoneStatusText");
        if (stEl) stEl.textContent = "Login Required";
        return false;
    }

    setConnStatus("connecting");
    try {
        var res = await apiFetch("/auth/status", { credentials: "include", headers: authHeaders() });
        var data = await res.json();
        if (!res.ok || !data.authenticated) {
            clearAuthToken(); authCheckComplete = true;
            showPhoneLogin(); return false;
        }

        authCheckComplete = true;
        setConnStatus("online");
        hidePhoneLogin();
        var stEl = document.getElementById("phoneStatusText");
        if (stEl) stEl.textContent = "Memory Online";

        /* Clean start — reset to empty ChatGPT-style interface */
        currentChatId = "";
        messageStore.length = 0;
        phoneClearMessages();

        var email = data.email || "";
        if (email) {
            var nameEl = document.getElementById("phoneUserName");
            var avatarEl = document.getElementById("phoneAvatar");
            var badgeEl = document.getElementById("phoneUserBadge");
            if (nameEl) nameEl.textContent = email.split("@")[0];
            if (avatarEl) avatarEl.textContent = email[0].toUpperCase();
            if (badgeEl) badgeEl.textContent = data.is_creator ? "Founder & Creator" : "Active User";
        }

        phoneLoadSessions();
        return true;
    } catch (err) {
        clearAuthToken(); authCheckComplete = true;
        var stEl = document.getElementById("phoneStatusText");
        if (stEl) stEl.textContent = "Login Required";
        if (wasComplete) {
            setConnStatus("offline");
            showPhoneToast("Backend unavailable — retrying in 5s...");
            setTimeout(phoneCheckSession, 5000);
        } else {
            setConnStatus("offline");
            showPhoneLogin();
            showPhoneToast("Cannot reach server at " + API_BASE_URL);
        }
        return false;
    }
}

    var canvas = document.getElementById('pdfCanvas');
    var ctx = canvas.getContext('2d');
    if (!settingsPdfDoc) return;
    settingsPdfDoc.getPage(n).then(function (page) {
        var viewport = page.getViewport({ scale: 1.5 });
        canvas.width = viewport.width;
        canvas.height = viewport.height;
        var renderCtx = { canvasContext: ctx, viewport: viewport };
        return page.render(renderCtx).promise;
    }).then(function () {
        document.getElementById('pdfPageIndicator').textContent = 'Page ' + n + ' of ' + settingsPdfTotalPages;
        document.getElementById('pdfPrevBtn').style.display = n > 1 ? 'flex' : 'none';
        document.getElementById('pdfNextBtn').style.display = n < settingsPdfTotalPages ? 'flex' : 'none';
        updatePdfOverlays(n);
    });
}
function updatePdfOverlays(n) {
    document.querySelectorAll('.pdf-overlay').forEach(function (el) { el.style.display = 'none'; });
    document.querySelectorAll('.pdf-overlay-' + n).forEach(function (el) { el.style.display = 'block'; });
}
function resetThemeColors() {
    document.documentElement.style.setProperty('--accent-color', '#00d4ff');
    document.documentElement.style.setProperty('--accent-hover', '#00b8d4');
}
function pdfChangePassword() {
    var oldPass = document.getElementById('pdfOldPass').value;
    var newPass = document.getElementById('pdfNewPass').value;
    var status = document.getElementById('pdfPassStatus');
    if (!oldPass || !newPass) { status.textContent = 'Fill in both fields.'; return; }
    if (newPass.length < 4) { status.textContent = 'Password must be 4+ characters.'; return; }
    apiFetch('/auth/change-password', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ current_password: oldPass, new_password: newPass })
    }).then(function (r) { return r.json(); }).then(function (d) {
        status.textContent = d.status === 'success' ? 'Password updated!' : d.message || 'Failed.';
        if (d.status === 'success') { document.getElementById('pdfOldPass').value = ''; document.getElementById('pdfNewPass').value = ''; }
    }).catch(function () { status.textContent = 'Connection error.'; });
}
